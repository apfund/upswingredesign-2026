Embeddable Hand Framework

Copy-paste HTML/CSS for embedding hand histories, board displays, and poker scenarios into any article. No dependencies — just CSS classes.

// Repeatable JS function to generate hand history blocks from data function createHandBlock(config) { const suitMap = { h: '♥', d: '♦', c: '♣', s: '♠' }; const suitClass = { h: 'heart', d: 'diamond', c: 'club', s: 'spade' }; function renderCard(rank, suit) { return `<div class="c-hand__card c-hand__card--${suitClass[suit]}"> <span class="c-hand__card-rank">${rank}</span> <span class="c-hand__card-suit">${suitMap[suit]}</span> </div>`; } function renderBoardCard(rank, suit) { return `<span class="c-hand__board-card c-hand__board-card--${suit}"> ${rank}${suitMap[suit]}</span>`; } // Usage: createHandBlock({ title: 'Cash Game', meta: 'NL500 6-max', badge: 'Key Hand', hero: { position: 'BTN', cards: [['A','h'],['K','s']], stack: '$512' }, streets: [ { name: 'Preflop', actions: [ { player: 'Hero', verb: 'raises to', amount: '$12', isHero: true }, { player: 'Villain', verb: '3-bets to', amount: '$42' }, ]}, { name: 'Flop', pot: '$85', board: [['K','s'],['9','h'],['4','d']], actions: [ { player: 'Villain', verb: 'checks' }, { player: 'Hero', verb: 'bets', amount: '$28', isHero: true }, ] }, ], result: 'What should Hero do?' }); }
